function toggleCodeMode() {
            const modal = document.getElementById('code-modal');
            if (!modal) {
                showToast && showToast('Chức năng Code chưa sẵn sàng', 2000);
                return;
            }
            modal.style.display = modal.style.display === 'none' || modal.style.display === '' ? 'flex' : 'none';
        }
